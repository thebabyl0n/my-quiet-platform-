async function sendMessage() { const input = document.getElementById('userInput'); const chat = document.getElementById('chat'); const text = input.value.trim(); if(text !== "") { // 1. نظهر رسالتك فوراً chat.innerHTML += `
أنت: ${text}
`; input.value = ""; chat.scrollTop = chat.scrollHeight; // 2. نبعت للسيرفر ونجيب الرد الحقيقي try { const response = await fetch('/chat', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({message: text}) }); const data = await response.json(); // 3. نظهر رد الظل الحقيقي اللي جاي من الـ API chat.innerHTML += `
الظل: ${data.reply}
`; } catch (error) { chat.innerHTML += `
الظل: فيه مشكلة حصلت يا حودة، اتأكد إنك رفعت الكود الجديد.
`; } chat.scrollTop = chat.scrollHeight; } }
