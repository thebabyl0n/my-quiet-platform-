// 1. تعديل دالة الإضافة للسلة عشان تسجل الإضافات صح function addToCart(k, ar, en, p, dept) { const s = document.getElementById('ex-'+k); const exEn = s.value; const exAr = s.options[s.selectedIndex].getAttribute('data-ar') || ""; const exP = parseFloat(s.options[s.selectedIndex].getAttribute('data-p') || 0); // تجهيز قائمة الإضافات كمصفوفة (Array) عشان شاشات الأقسام تشوفها let extrasArray = exAr ? [exAr] : []; const existingItem = cart.find(i => i.k === k && i.exEn === exEn); if (existingItem) { existingItem.qty += 1; } else { cart.push({ k, ar, en, p, exAr, exEn, exP, extras: extrasArray, // الخانة الجديدة اللي كانت ناقصة! qty: 1, dept: dept }); } talk(en + (exEn ? " with " + exEn : "")); renderCart(); } // 2. تعديل دالة الإرسال (اختياري لزيادة التأكيد) function submitOrder() { if(!cart.length) return alert("السلة فارغة"); const finalTotal = document.getElementById('total').innerText; const subTotal = cart.reduce((acc, i) => acc + (i.p + i.exP) * i.qty, 0); let dAmount = (discountType === "percent") ? (subTotal * discountValReal / 100) : discountValReal; const notes = document.getElementById('notes').value; // تجهيز الداتا النهائية const orderData = { items: cart, type: oType, table: tNum || 'سفري', notes: notes, promoUsed: activePromo, discountAmount: dAmount.toFixed(2), totalPrice: finalTotal, status: 'قيد التحضير ⏳', time: new Date().toLocaleTimeString('ar-SA'), // التأكد من تصفير حالات الأقسام baristaReady: false, kitchenReady: false, dessertReady: false }; const orderRef = database.ref('live_orders').push(); currentOrderId = orderRef.key; orderRef.set(orderData); database.ref('orders/' + currentOrderId).set(orderData); // ... (باقي كود الحالة والواتساب زي ما هو) document.getElementById('order-status-bar').style.display = 'block'; database.ref('orders/' + currentOrderId + '/status').on('value', s => { const status = s.val(); document.getElementById('order-status-bar').innerText = "حالة الطلب: " + status; if(status === 'جاهز للاستلام ✅') { showToast(successMsg || "طلبك جاهز!"); if(readyAudio) talk(readyAudio); } }); let msg = `*طلب جديد* ☕\n*المكان*: ${oType} ${tNum?'[طاولة '+tNum+']':''}\n`; cart.forEach(i => { msg += `• ${i.qty}x ${i.ar} ${i.exAr ? '[+'+i.exAr+']' : ''}\n`; }); msg += `*الإجمالي*: ${finalTotal} ريال\n*الملاحظات*: ${notes}`; window.open(`https://wa.me/966579747707?text=${encodeURIComponent(msg)}`); cart = []; renderCart(); activePromo = ""; discountValReal = 0; showToast("تم إرسال الطلب بنجاح! 🎉"); }
